Project rent {
  database_type: 'PostgreSQL'
}

table users {
  id integer [primary key]
  email varchar [unique, not null]
  password varchar [not null]
  address_id integer
  points integer // cache
  created_at datetime
  last_login_at datetime
  newsletter_started_at datetime
  internal_notes varchar
}

table categories {
  id integer [primary key]
  name string
  slug string
}

table addresses {
  id integer [primary key]
  last_name varchar
  second_name varchar
  first_line varchar
  second_line varchar
  postal_code varchar
  city varchar
  country varchar
}

table orders {
  id integer [primary key]
  user_id integer [not null]
  created_at datetime
  status status [default: "PENDING"]
  payment_code uuid
  used_points bool [default: false]
}

table points_history {
  id integer [primary key]
  order_id integer [not null]
  points integer
}

enum status {
  "PENDING"
  "UNPAID"
  "PAID"
  "GIVEN_OUT"
  "FINISHED"
  "CANCELLED"
}

table order_instances {
  id integer [primary key]
  order_id integer [not null]
  instance_id integer [not null]
  startDate date
  endDate date
  price double // per day

  checks {
    `startDate <= endDate`
    `price > 0`
  }
}

table products {
  id integer [primary key]
  name varchar [not null]
  description text
  price double
  slug varchar [unique]
  meta_description varchar
  visibility_status bool
  category_id integer

  checks {
    `price > 0`
  }
}

table product_images {
  id integer [primary key]
  product_id integer [not null]
  image image
  alt_text varchar
  display_order tinyint

  indexes {
    (product_id, display_order) [unique]
  }
}

table product_sizes {
  id integer [primary key]
  product_id integer
  size varchar
  description text

  indexes {
    (product_id, size) [unique]
  }
}

table favorites {
  id integer [primary key]
  user_id integer [not null]
  product_id integer [not null]
}

enum instance_status {
  "AVAILABLE"
  "MAINTENANCE"
  "BROKEN"
  "RETIRED"
}

table instances {
  id integer [primary key]
  product_id integer [not null]
  status instance_status [default: "AVAILABLE"]
  size varchar
}

table cart_items {
  id integer [primary key]
  user_id integer [not null]
  product_id integer [not null]
  product_size_id integer
  quantity integer
  start_date date
  end_date date
  created_at date

  checks {
    `start_date <= end_date`
    `quantity > 0`
  }
}

table calendar_blocks {
  id integer [primary key]
  instance_id integer [not null]
  blocked_from date
  blocked_until date
  block_reason varchar

  checks {
    `blocked_from <= blocked_until`
  }
}

ref: products.id < favorites.product_id

ref: products.id < product_images.product_id
ref: orders.id < points_history.order_id
ref: users.id < cart_items.user_id
ref: instances.id < calendar_blocks.instance_id
ref: products.id < product_sizes.product_id
