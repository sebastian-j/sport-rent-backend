table users {
  id integer [primary key]
  email email [unique, not null]
  password varchar [not null]
  address_id integer
  points integer // cache
  created_at datetime
  last_login datetime
}

table addresses {
  id integer [primary key]
  building varchar
  street varchar
  postal_code varchar
  city varchar
  country varchar
}

table orders {
  id integer [primary key]
  user_id integer [not null]
  created_at datetime
  status status [default: "PENDING"]
  payment_code uuid
}

table points_history {
  id integer [primary key]
  user_id integer [not null]
  order_id integer [not null]
  points integer
}

enum status {
  "PENDING"
  "UNPAID"
  "PAID"
  "GIVEN_OUT"
  "FINISHED"
  "CANCELLED"
}

table order_instances {
  id integer [primary key]
  order_id integer [not null]
  instance_id integer [not null]
  startDate date
  endDate date
  price double // per day
}

table products {
  id integer [primary key]
  name varchar [not null]
  description text
  price double
}

table product_images {
  id integer [primary key]
  product_id integer [not null]
  image image
}

table favorites {
  id integer [primary key]
  user_id integer [not null]
  product_id integer [not null]
}

enum instance_status {
  "AVAILABLE"
  "MAINTENANCE"
  "BROKEN"
  "RETIRED"
}

table instances {
  id integer [primary key]
  product_id integer [not null]
  status instance_status [default: "AVAILABLE"]
}

ref: users.id < orders.user_id
ref: orders.id < order_instances.order_id
ref: instances.id < order_instances.instance_id
ref: products.id < instances.product_id
ref: users.id < favorites.user_id
ref: products.id < favorites.product_id
ref: addresses.id - users.address_id
ref: products.id < product_images.product_id
ref: users.id < points_history.user_id
ref: orders.id < points_history.order_id
